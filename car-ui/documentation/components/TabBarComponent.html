import { Component } from "@angular/core";
import { IsActiveMatchOptions } from "@angular/router";

@Component({
  selector: "app-tab-bar",
  templateUrl: "./tab-bar.component.html",
  styleUrls: ["./tab-bar.component.scss"],
})
export class TabBarComponent {
  matchOptions: IsActiveMatchOptions = {
    paths: "exact",
    matrixParams: "exact",
    queryParams: "subset",
    fragment: "ignored",
  };
  shrinkTabs: { id: number; tabName: string; url: string }[] = [
    { id: 0, tabName: "Azure", url: "/dashboard" },
    { id: 1, tabName: "GCP", url: "/dashboard/gcp" },
  ];
}

<section class="shrink-tabs mb-5">
  <div>
    <nav mat-tab-nav-bar color="primary" [tabPanel]="tabPanel">
      <a
        mat-tab-link
        *ngFor="let tabItem of shrinkTabs"
        [routerLink]="tabItem.url"
        routerLinkActive
        #rla="routerLinkActive"
        [routerLinkActiveOptions]="matchOptions"
        [active]="rla.isActive"
      >
        <span id="navstepstext">{{ tabItem.tabName }}</span>
      </a>
    </nav>
    <mat-tab-nav-panel #tabPanel class="tab-body">
      <router-outlet></router-outlet>
    </mat-tab-nav-panel>
  </div>
</section>
