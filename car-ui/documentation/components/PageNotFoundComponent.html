import { Component } from "@angular/core";

  styleUrls: ["./page-not-found.component.scss"],
})
export class PageNotFoundComponent {
  constructor(private router: Router) {}

  goToDashboard() {
    this.router.navigate(["/dashboard"]);
  }
}
